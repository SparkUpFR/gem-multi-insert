def execute
  result = nil
  ActiveRecord::Base.connection_pool.with_connection do |con|
    result = con.execute(to_sql)
  end
  if @sql_returning.nil?
    nil
  else
    if @returning_flat
      result.values.map{|r| r.first}
    else
      result
    end
  end
end
